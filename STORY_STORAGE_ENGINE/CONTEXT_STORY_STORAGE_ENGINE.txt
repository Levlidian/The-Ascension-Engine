const modifier = (text, stop) => {
  const locationBlock = getLocationBlock(state.storyEngine.currentLocation);
  const characterBlock = getCharacterBlock();
  const escalationBlock = `
ESCALATION LEVEL: ${state.storyEngine.escalation} / 5
`;

  text =
    locationBlock +
    characterBlock +
    escalationBlock +
    "\n---\n" +
    text;

  if (typeof InnerSelfContext === "function") {
    [text, stop] = InnerSelfContext(text, stop);
  }

  return { text, stop };
};

modifier(text, stop);
